    func binarySearch(arr:[Int],searchNum:Int)->Bool{
        if arr.count == 1{
            return arr.first! == searchNum ? true : false
        }
        
        let mid = arr[arr.count/2]
        
        if mid > searchNum{
            return binarySearch(arr: arr.filter{ $0 < mid }, searchNum: searchNum)
        }else{
            return binarySearch(arr: arr.filter{ $0 > mid }, searchNum: searchNum)
        }
    }
